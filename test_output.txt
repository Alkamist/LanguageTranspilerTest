fn add_numbers(x: f64, y: f64) -> f64 {
    x + y
}

fn main() {
    let mut x = 10.0;
    let mut y = 10.0;

    fn subtract_numbers(x: f64, y: f64) -> f64 {
        x - y
    }

    for i in 0..10 {
        for j in 0..10 {
            x = add_numbers(x, 1);
            y = subtract_numbers(x, 1);
        }
    }

    let mut z = 5;

    if z == 5 {
        x = 5.0;
    }
    else if z == 3 {
        x = 3.0;
    }
    else {
        x = 10.0;
    }
}
